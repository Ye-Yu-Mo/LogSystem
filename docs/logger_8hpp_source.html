548 bool hasLogger(const std::string &name)
549 {
550 std::unique_lock<std::mutex> lock(_mutex);
551 auto it = _loggers.find(name);
552 if (it == _loggers.end())
553 return false;
554 return true;
555 }
562 Logger::ptr getLogger(const std::string &name)
563 {
564 std::unique_lock<std::mutex> lock(_mutex);
565 auto it = _loggers.find(name);
566 if (it == _loggers.end())

568 std::cout << "未找到日志器！日志器名称："<<name<<std::endl;
