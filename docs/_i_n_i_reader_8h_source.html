15// Read an INI file into easy-to-access name/value pairs. (Note that I've gone
16// for simplicity here rather than speed, but it should be pretty decent.)
17class INIReader
18{
19public:
20 // Construct INIReader and parse given filename. See ini.h for more info
21 // about the parsing.
22 INIReader(std::string filename);
23 ~INIReader();
24
25 // Return the result of ini_parse(), i.e., 0 on success, line number of
26 // first error on parse error, or -1 on file open error.
27 int ParseError();
28
29 // Get a string value from INI file, returning default_value if not found.
30 std::string Get(std::string section, std::string name,
31 std::string default_value);
32
33 // Get an integer (long) value from INI file, returning default_value if
34 // not found or not a valid integer (decimal "1234", "-1234", or hex "0x4d2").
35 long GetInteger(std::string section, std::string name, long default_value);
36
37 // Get a real (floating point double) value from INI file, returning
38 // default_value if not found or not a valid floating point value
39 // according to strtod().
40 double GetReal(std::string section, std::string name, double default_value);
41
42 // Get a boolean value from INI file, returning default_value if not found or if
43 // not a valid true/false value. Valid true values are "true", "yes", "on", "1",
44 // and valid false values are "false", "no", "off", "0" (not case sensitive).
45 bool GetBoolean(std::string section, std::string name, bool default_value);
46
47 // Returns all the section names from the INI file, in alphabetical order, but in the
48 // original casing
49 std::set<std::string> GetSections() const;
50
51 // Returns all the field names from a section in the INI file, in alphabetical order,
52 // but in the original casing. Returns an empty set if the field name is unknown
53 std::set<std::string> GetFields(std::string section) const;
54
55private:
56 int _error;
57 std::map<std::string, std::string> _values;
58 // Because we want to retain the original casing in _fields, but
59 // want lookups to be case-insensitive, we need both _fields and _values
60 std::set<std::string> _sections;
61 std::map<std::string, std::set<std::string>*> _fields;
62 static std::string MakeKey(std::string section, std::string name);
63 static int ValueHandler(void* user, const char* section, const char* name,
64 const char* value);
65};
